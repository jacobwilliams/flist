    impure recursive subroutine destroy_node(me)

    implicit none

    type(node),pointer :: me

    if (associated(me)) then
        call me%destroy()
        call destroy_node(me%next)
        nullify(me%previous)
        deallocate(me)
        nullify(me)
    end if

    end subroutine destroy_node
